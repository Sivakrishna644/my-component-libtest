## Install

```bash
npm install --save my-component-libtest
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'my-component-libtest'
import 'my-component-libtest/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
